CREATE TABLE app_users ( id INTEGER PRIMARY KEY, name TEXT NOT NULL, married BOOLEAN NOT NULL DEFAULT 0, birth_date DATE NOT NULL, last_login DATETIME NOT NULL );

INSERT INTO app_users (id, name, married, birth_date, last_login) VALUES (1, 'Dana', TRUE, '2004-06-15', '2026-04-10 09:10:00'), (2, 'Omar', FALSE, '2003-11-02', '2026-03-20 18:40:00'), (3, 'Noa', TRUE, '2005-01-30', '2026-04-12 07:55:00');

3

Solution — Constraints

Step 1 — Create table with all required constraints

CREATE TABLE exam_results ( id INTEGER PRIMARY KEY NOT NULL, email TEXT NOT NULL UNIQUE, student_name TEXT NOT NULL, grade INTEGER CHECK (grade >= 0 AND grade <= 100), status TEXT DEFAULT 'pending' );

Step 2 — Insert all 3 rows

INSERT INTO exam_results (id, email, student_name, grade, status) VALUES (1, 'dana@mail.com', 'Dana', 92, 'approved'), (2, 'omar@mail.com', 'Omar', 76, 'pending'), (3, 'noa@mail.com', 'Noa', 85, 'approved');

Step 3a — Find names of approved students

SELECT student_name FROM exam_results WHERE status = 'approved';
Result:
student_name
Dana
Noa

Step 3b — Find users where student_name AND email both start with 'd'

SELECT * FROM exam_results WHERE student_name LIKE 'd%' AND email LIKE 'd%';
Result:
idemailstudent_namegradestatus
1dana@mail.comDana92approved

LIKE is case-insensitive in SQLite by default for ASCII letters, so 'd%' matches 'Dana' and 'dana@mail.com'
