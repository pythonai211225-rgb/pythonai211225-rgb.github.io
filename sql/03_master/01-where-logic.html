AND / OR / NOT logic

Examples
-- Action movies after 2010
SELECT title, year, rating FROM movies
WHERE genre = 'Action' AND year > 2010;

-- Drama OR Family
SELECT title, genre FROM movies
WHERE genre = 'Drama' OR genre = 'Family';

-- All movies that are NOT Horror
SELECT title, genre FROM movies
WHERE NOT genre = 'Horror';

Precedence and parentheses

Without parentheses, AND is evaluated before OR.

Compare
-- can return unexpected rows
SELECT title FROM movies
WHERE genre = 'Action' OR genre = 'Drama' AND rating > 7.5;

-- clear intent using parentheses
SELECT title FROM movies
WHERE (genre = 'Action' OR genre = 'Drama') AND rating > 7.5;
