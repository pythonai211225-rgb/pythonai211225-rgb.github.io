JOIN / GROUP BY

B SELECT GROUP BY customer_id FROM orders;
C SELECT customer_id FROM orders ORDER BY customer_id;

Which INNER JOIN correctly connects orders to customers?
A SELECT * FROM orders INNER JOIN customers ON orders.customer_id = customers.id;

B SELECT city, COUNT(*) FROM customers GROUP city JOIN orders;

D SELECT city FROM customers JOIN COUNT(orders);

Hard Questions

CASE WHEN / COALESCE / SUBQUERIES
Hard · Q1
Which query replaces NULL city with 'Unknown' for all customers?
A SELECT name, city OR 'Unknown' FROM customers;
B SELECT name, REPLACE(city, NULL, 'Unknown') FROM customers;
C SELECT name, COALESCE(city, 'Unknown') AS city FROM customers;
D SELECT name, IFNULL city = 'Unknown' FROM customers;
Hard · Q2
Which subquery returns products priced above the average price?
A SELECT * FROM products WHERE price > (SELECT AVG(price) FROM products);
B SELECT AVG(price) FROM products WHERE price > products;
C SELECT * FROM products ABOVE AVG(price);
D SELECT * FROM products WHERE AVG(price) > price;
Hard · Q3
Which query finds employees whose salary is the maximum in the table?
A SELECT * FROM employees WHERE salary = MAX(salary);
B SELECT * FROM employees WHERE salary = (SELECT MAX(salary) FROM employees);
C SELECT MAX(salary) FROM employees WHERE employee = *;
D SELECT * FROM employees HAVING salary = MAX(salary);
Hard · Q4
Which query labels products as 'Expensive' if price > 100, else 'Cheap'?
A SELECT name, WHEN price > 100 THEN 'Expensive' ELSE 'Cheap' FROM products;
B SELECT name, CASE WHEN price > 100 THEN 'Expensive' ELSE 'Cheap' END AS label FROM products;
C SELECT name, IF(price > 100) THEN 'Expensive' ELSE 'Cheap' FROM products;
D SELECT name, price CASE 'Expensive' WHEN > 100 FROM products;
Hard · Q5
Which correlated subquery keeps orders above that customer's average order total?
A SELECT * FROM orders o WHERE total > (SELECT AVG(total) FROM orders);

Which query shows students who scored above their class average?
A SELECT * FROM students WHERE score > AVG(score);
B SELECT * FROM students s WHERE score > (SELECT AVG(score) FROM students);
C SELECT * FROM students s WHERE score > (SELECT AVG(score) FROM students WHERE class = s.class);

Hard · Q7
Which query counts how many orders have total > 500 per customer?
A SELECT customer_id, COUNT(*) WHERE total > 500 FROM orders GROUP BY customer_id;
B SELECT customer_id, COUNT(total > 500) FROM orders GROUP BY customer_id;
C SELECT customer_id, COUNT(CASE WHEN total > 500 THEN 1 END) AS big_orders FROM orders GROUP BY customer_id;
D SELECT customer_id, SUM(CASE WHEN total) > 500 FROM orders GROUP BY customer_id;
Hard · Q8
Which query shows salary level: 'High' if > 8000, 'Mid' if > 4000, else 'Low'?
A SELECT name, CASE salary > 8000 THEN 'High' WHEN > 4000 THEN 'Mid' ELSE 'Low' FROM employees;
B SELECT name, CASE WHEN salary > 8000 THEN 'High' WHEN salary > 4000 THEN 'Mid' ELSE 'Low' END AS level FROM employees;
C SELECT name, IF(salary > 8000, 'High', IF(salary > 4000, 'Mid', 'Low')) FROM employees;
D SELECT name, SWITCH salary CASE 8000 THEN 'High' CASE 4000 THEN 'Mid' ELSE 'Low' FROM employees;

B SELECT id, total, discount = 0 WHERE discount IS NULL FROM orders;
C SELECT id, total, COALESCE(discount, 0) AS discount FROM orders;
D SELECT id, total, REPLACE(discount, NULL, 0) FROM orders;
Hard · Q10
Which query counts paid and unpaid orders in a single result row?
A SELECT COUNT(status = 'paid') AS paid, COUNT(status = 'unpaid') AS unpaid FROM orders;
B SELECT SUM(paid) AS paid, SUM(unpaid) AS unpaid FROM orders GROUP BY status;
C SELECT COUNT(*) FILTER (WHERE status = 'paid') AS paid, COUNT(*) FILTER (WHERE status = 'unpaid') AS unpaid FROM orders;
D SELECT COUNT(CASE WHEN status = 'paid' THEN 1 END) AS paid, COUNT(CASE WHEN status = 'unpaid' THEN 1 END) AS unpaid FROM orders;
